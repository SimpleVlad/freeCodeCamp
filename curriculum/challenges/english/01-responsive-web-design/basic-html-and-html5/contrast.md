function contrast() {
  let mat = cv.imread("imageSrc");
  let dst = new cv.Mat();
  let cont = Number(document.getElementById("contrast").value);
  let brig = Number(document.getElementById("brightness").value);
  cv.convertScaleAbs(mat, dst, cont, brig);
  cv.imshow("canvasOutput", dst);
  mat.delete();
  dst.delete();
};
